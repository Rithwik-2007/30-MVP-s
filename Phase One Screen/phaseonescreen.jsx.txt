import React, { useState, useEffect, useMemo } from 'react';
import { 
  Users, 
  CheckCircle, 
  XCircle, 
  Upload, 
  FileText, 
  Settings, 
  Search, 
  Trash2, 
  UserPlus, 
  ChevronRight,
  AlertCircle
} from 'lucide-react';

// --- Components ---

const Button = ({ children, onClick, variant = 'primary', className = '', disabled = false }) => {
  const baseStyle = "px-4 py-2 rounded-md font-medium transition-colors focus:outline-none focus:ring-2 focus:ring-offset-1";
  const variants = {
    primary: "bg-blue-600 text-white hover:bg-blue-700 focus:ring-blue-500",
    secondary: "bg-white text-gray-700 border border-gray-300 hover:bg-gray-50 focus:ring-gray-400",
    success: "bg-green-600 text-white hover:bg-green-700 focus:ring-green-500",
    danger: "bg-red-600 text-white hover:bg-red-700 focus:ring-red-500",
    ghost: "bg-transparent text-gray-600 hover:bg-gray-100 focus:ring-gray-400",
  };

  return (
    <button 
      onClick={onClick} 
      disabled={disabled}
      className={`${baseStyle} ${variants[variant]} ${disabled ? 'opacity-50 cursor-not-allowed' : ''} ${className}`}
    >
      {children}
    </button>
  );
};

const Card = ({ children, className = '' }) => (
  <div className={`bg-white rounded-lg border border-gray-200 shadow-sm ${className}`}>
    {children}
  </div>
);

const Badge = ({ children, type }) => {
  const styles = {
    success: "bg-green-100 text-green-800",
    warning: "bg-yellow-100 text-yellow-800",
    error: "bg-red-100 text-red-800",
    neutral: "bg-gray-100 text-gray-800",
  };
  return (
    <span className={`px-2 py-0.5 rounded text-xs font-medium ${styles[type] || styles.neutral}`}>
      {children}
    </span>
  );
};

// --- Main Application ---

export default function App() {
  // --- State ---
  const [view, setView] = useState('dashboard'); // 'dashboard', 'setup', 'add'
  const [activeTab, setActiveTab] = useState('pending'); // 'pending', 'selected', 'rejected'
  
  // Job Criteria
  const [jobTitle, setJobTitle] = useState(() => localStorage.getItem('hr_jobTitle') || 'Software Engineer');
  const [requiredSkills, setRequiredSkills] = useState(() => {
    const saved = localStorage.getItem('hr_skills');
    return saved ? JSON.parse(saved) : ['React', 'JavaScript', 'Node.js'];
  });
  
  // Candidates
  const [candidates, setCandidates] = useState(() => {
    const saved = localStorage.getItem('hr_candidates');
    return saved ? JSON.parse(saved) : [];
  });

  // Inputs for Setup
  const [newSkill, setNewSkill] = useState('');
  
  // Inputs for Adding Candidate
  const [inputText, setInputText] = useState('');
  const [dragActive, setDragActive] = useState(false);

  // --- Effects ---
  useEffect(() => {
    localStorage.setItem('hr_jobTitle', jobTitle);
    localStorage.setItem('hr_skills', JSON.stringify(requiredSkills));
    localStorage.setItem('hr_candidates', JSON.stringify(candidates));
  }, [jobTitle, requiredSkills, candidates]);

  // --- Logic ---

  const analyzeCandidate = (text, nameOverride = null) => {
    const lines = text.split('\n').map(l => l.trim()).filter(l => l.length > 0);
    // Simple heuristic: Name is often the first non-empty line
    const name = nameOverride || lines[0] || "Unknown Candidate";
    
    // Normalize text for search
    const lowerText = text.toLowerCase();
    
    // Check skills
    const foundSkills = requiredSkills.filter(skill => 
      lowerText.includes(skill.toLowerCase())
    );

    const matchCount = foundSkills.length;
    const totalCount = requiredSkills.length;
    const percentage = totalCount === 0 ? 100 : (matchCount / totalCount);

    let status = 'Does Not Meet';
    let statusType = 'error';

    if (percentage === 1) {
      status = 'Meets Requirements';
      statusType = 'success';
    } else if (percentage >= 0.5) {
      status = 'Partial Match';
      statusType = 'warning';
    }

    return {
      id: Date.now().toString() + Math.random().toString(),
      name,
      text, // Store original text if needed for viewing
      foundSkills,
      missingSkills: requiredSkills.filter(s => !foundSkills.includes(s)),
      status,
      statusType,
      matchPercentage: Math.round(percentage * 100),
      decision: 'pending', // 'pending', 'selected', 'rejected'
      dateAdded: new Date().toISOString()
    };
  };

  const handleAddSkill = (e) => {
    e.preventDefault();
    if (newSkill.trim() && !requiredSkills.includes(newSkill.trim())) {
      setRequiredSkills([...requiredSkills, newSkill.trim()]);
      setNewSkill('');
    }
  };

  const removeSkill = (skillToRemove) => {
    setRequiredSkills(requiredSkills.filter(s => s !== skillToRemove));
  };

  const processFile = async (file) => {
    if (file.type === 'text/plain') {
      const text = await file.text();
      const candidate = analyzeCandidate(text);
      setCandidates(prev => [candidate, ...prev]);
      setView('dashboard');
    } else {
      alert("For this phase-one version, please upload .txt files only or paste the text content.");
    }
  };

  const handleDrop = (e) => {
    e.preventDefault();
    e.stopPropagation();
    setDragActive(false);
    if (e.dataTransfer.files && e.dataTransfer.files[0]) {
      processFile(e.dataTransfer.files[0]);
    }
  };

  const handleManualAdd = () => {
    if (!inputText.trim()) return;
    const candidate = analyzeCandidate(inputText);
    setCandidates(prev => [candidate, ...prev]);
    setInputText('');
    setView('dashboard');
  };

  const handleDecision = (id, decision) => {
    setCandidates(candidates.map(c => 
      c.id === id ? { ...c, decision } : c
    ));
  };

  const deleteCandidate = (id) => {
    setCandidates(candidates.filter(c => c.id !== id));
  };

  const clearAllCandidates = () => {
    if (confirm('Are you sure you want to clear all candidates?')) {
      setCandidates([]);
    }
  };

  // --- Filtered Views ---
  const displayedCandidates = useMemo(() => {
    return candidates.filter(c => c.decision === activeTab);
  }, [candidates, activeTab]);

  const stats = useMemo(() => {
    return {
      pending: candidates.filter(c => c.decision === 'pending').length,
      selected: candidates.filter(c => c.decision === 'selected').length,
      rejected: candidates.filter(c => c.decision === 'rejected').length,
    };
  }, [candidates]);


  // --- Render Views ---

  if (view === 'setup') {
    return (
      <div className="min-h-screen bg-gray-50 flex flex-col items-center py-12 px-4 sm:px-6 lg:px-8">
        <div className="max-w-md w-full space-y-8">
          <div>
            <h2 className="mt-6 text-center text-3xl font-extrabold text-gray-900">
              Phase One Criteria
            </h2>
            <p className="mt-2 text-center text-sm text-gray-600">
              Define the baseline requirements for the {jobTitle} role.
            </p>
          </div>
          
          <Card className="p-6 space-y-6">
            <div>
              <label className="block text-sm font-medium text-gray-700">Job Role Title</label>
              <input
                type="text"
                value={jobTitle}
                onChange={(e) => setJobTitle(e.target.value)}
                className="mt-1 block w-full border border-gray-300 rounded-md shadow-sm py-2 px-3 focus:outline-none focus:ring-blue-500 focus:border-blue-500 sm:text-sm"
              />
            </div>

            <div>
              <label className="block text-sm font-medium text-gray-700">Required Skills / Keywords</label>
              <div className="flex mt-1">
                <input
                  type="text"
                  value={newSkill}
                  onChange={(e) => setNewSkill(e.target.value)}
                  onKeyDown={(e) => e.key === 'Enter' && handleAddSkill(e)}
                  placeholder="e.g. React, Python, Leadership"
                  className="flex-1 block w-full border border-gray-300 rounded-l-md shadow-sm py-2 px-3 focus:outline-none focus:ring-blue-500 focus:border-blue-500 sm:text-sm"
                />
                <button
                  onClick={handleAddSkill}
                  className="inline-flex items-center px-4 py-2 border border-l-0 border-gray-300 rounded-r-md bg-gray-50 text-gray-500 text-sm hover:bg-gray-100"
                >
                  Add
                </button>
              </div>
              <p className="text-xs text-gray-500 mt-1">Press Enter to add.</p>
            </div>

            <div className="flex flex-wrap gap-2">
              {requiredSkills.map(skill => (
                <span key={skill} className="inline-flex items-center px-2.5 py-0.5 rounded-full text-sm font-medium bg-blue-100 text-blue-800">
                  {skill}
                  <button onClick={() => removeSkill(skill)} className="ml-1.5 inline-flex items-center justify-center text-blue-400 hover:text-blue-600">
                    Ã—
                  </button>
                </span>
              ))}
              {requiredSkills.length === 0 && (
                <span className="text-sm text-gray-400 italic">No skills defined yet.</span>
              )}
            </div>
          </Card>

          <div className="flex justify-between">
             <Button variant="ghost" onClick={() => setView('dashboard')}>Cancel</Button>
             <Button onClick={() => setView('dashboard')}>Save & Continue</Button>
          </div>
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-50 text-gray-900 font-sans">
      {/* Header */}
      <header className="bg-white border-b border-gray-200 sticky top-0 z-10">
        <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8">
          <div className="flex justify-between h-16 items-center">
            <div className="flex items-center">
              <Users className="h-6 w-6 text-blue-600 mr-2" />
              <h1 className="text-xl font-bold text-gray-900">PhaseOne Filter</h1>
              <span className="ml-4 px-3 py-1 bg-gray-100 rounded-full text-sm text-gray-600 font-medium">
                {jobTitle}
              </span>
            </div>
            <div className="flex items-center space-x-4">
              <Button variant="ghost" className="text-sm" onClick={() => setView('setup')}>
                <Settings className="h-4 w-4 mr-1 inline" /> Criteria
              </Button>
              <Button variant="primary" onClick={() => setView('add')} className="flex items-center">
                <UserPlus className="h-4 w-4 mr-2" /> Add Candidate
              </Button>
            </div>
          </div>
        </div>
      </header>

      <main className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 py-8">
        
        {/* Add Candidate Modal/Overlay Area */}
        {view === 'add' && (
           <div className="mb-8 animate-fade-in">
             <Card className="p-6 border-dashed border-2 border-blue-200 bg-blue-50">
               <div className="flex justify-between items-start mb-4">
                  <h3 className="text-lg font-medium text-blue-900">Add New Candidate</h3>
                  <button onClick={() => setView('dashboard')} className="text-blue-500 hover:text-blue-700">Close</button>
               </div>
               
               <div 
                 className={`flex flex-col items-center justify-center border-2 border-dashed rounded-lg p-12 transition-colors ${dragActive ? 'border-blue-500 bg-blue-100' : 'border-gray-300 bg-white'}`}
                 onDragEnter={(e) => { e.preventDefault(); e.stopPropagation(); setDragActive(true); }}
                 onDragLeave={(e) => { e.preventDefault(); e.stopPropagation(); setDragActive(false); }}
                 onDragOver={(e) => { e.preventDefault(); e.stopPropagation(); }}
                 onDrop={handleDrop}
               >
                 <Upload className="h-12 w-12 text-gray-400 mb-4" />
                 <p className="text-gray-600 mb-2 font-medium">Drag & drop a text file (.txt)</p>
                 <p className="text-gray-400 text-sm mb-4">- OR -</p>
                 <textarea 
                    className="w-full max-w-2xl p-4 border rounded-md text-sm font-mono h-48 focus:ring-2 focus:ring-blue-500 focus:outline-none"
                    placeholder="Paste resume text here..."
                    value={inputText}
                    onChange={(e) => setInputText(e.target.value)}
                 />
                 <div className="mt-4 flex gap-2">
                   <label className="cursor-pointer">
                      <span className="px-4 py-2 rounded-md bg-white border border-gray-300 text-gray-700 hover:bg-gray-50 text-sm font-medium">
                        Browse Files
                      </span>
                      <input type="file" className="hidden" accept=".txt" onChange={(e) => e.target.files[0] && processFile(e.target.files[0])} />
                   </label>
                   <Button onClick={handleManualAdd} disabled={!inputText.trim()}>
                     Analyze Text
                   </Button>
                 </div>
               </div>
             </Card>
           </div>
        )}

        {/* Dashboard Tabs */}
        <div className="flex space-x-1 mb-6 bg-white p-1 rounded-lg border border-gray-200 w-fit">
          <button
            onClick={() => setActiveTab('pending')}
            className={`px-4 py-2 rounded-md text-sm font-medium transition-all ${activeTab === 'pending' ? 'bg-blue-50 text-blue-700 shadow-sm' : 'text-gray-500 hover:text-gray-700'}`}
          >
            Review Needed <span className="ml-2 bg-blue-200 text-blue-800 py-0.5 px-2 rounded-full text-xs">{stats.pending}</span>
          </button>
          <button
            onClick={() => setActiveTab('selected')}
            className={`px-4 py-2 rounded-md text-sm font-medium transition-all ${activeTab === 'selected' ? 'bg-green-50 text-green-700 shadow-sm' : 'text-gray-500 hover:text-gray-700'}`}
          >
            Selected <span className="ml-2 bg-green-200 text-green-800 py-0.5 px-2 rounded-full text-xs">{stats.selected}</span>
          </button>
          <button
            onClick={() => setActiveTab('rejected')}
            className={`px-4 py-2 rounded-md text-sm font-medium transition-all ${activeTab === 'rejected' ? 'bg-red-50 text-red-700 shadow-sm' : 'text-gray-500 hover:text-gray-700'}`}
          >
            Rejected <span className="ml-2 bg-red-200 text-red-800 py-0.5 px-2 rounded-full text-xs">{stats.rejected}</span>
          </button>
        </div>

        {/* Candidate List */}
        <div className="space-y-4">
          {displayedCandidates.length === 0 ? (
            <div className="text-center py-12 bg-white rounded-lg border border-dashed border-gray-300">
              <div className="mx-auto h-12 w-12 text-gray-400 flex items-center justify-center rounded-full bg-gray-100 mb-4">
                <Search className="h-6 w-6" />
              </div>
              <h3 className="mt-2 text-sm font-medium text-gray-900">No candidates found</h3>
              <p className="mt-1 text-sm text-gray-500">
                {activeTab === 'pending' ? "Add candidates to start screening." : `No candidates in the ${activeTab} list.`}
              </p>
              {activeTab === 'pending' && (
                <div className="mt-6">
                  <Button onClick={() => setView('add')}>Add Candidate</Button>
                </div>
              )}
            </div>
          ) : (
            displayedCandidates.map((candidate) => (
              <Card key={candidate.id} className="p-6 transition-shadow hover:shadow-md">
                <div className="flex flex-col md:flex-row md:items-start md:justify-between gap-4">
                  
                  {/* Info Section */}
                  <div className="flex-1 min-w-0">
                    <div className="flex items-center mb-2">
                       <div className="h-10 w-10 rounded-full bg-blue-100 flex items-center justify-center text-blue-700 font-bold text-lg mr-3">
                         {candidate.name.charAt(0).toUpperCase()}
                       </div>
                       <div>
                         <h3 className="text-lg font-bold text-gray-900 truncate">{candidate.name}</h3>
                         <p className="text-xs text-gray-500">Added: {new Date(candidate.dateAdded).toLocaleDateString()}</p>
                       </div>
                    </div>

                    <div className="mt-4 grid grid-cols-1 sm:grid-cols-2 gap-4">
                        <div>
                          <p className="text-xs font-semibold text-gray-500 uppercase tracking-wide mb-2">Requirements Check</p>
                          <div className="flex items-center space-x-2">
                             <Badge type={candidate.statusType}>{candidate.status}</Badge>
                             <span className="text-sm text-gray-500">({candidate.matchPercentage}% match)</span>
                          </div>
                        </div>
                        
                        <div>
                          <p className="text-xs font-semibold text-gray-500 uppercase tracking-wide mb-2">Skills Found</p>
                          <div className="flex flex-wrap gap-1">
                            {candidate.foundSkills.length > 0 ? (
                              candidate.foundSkills.map(skill => (
                                <span key={skill} className="px-2 py-0.5 rounded text-xs bg-green-50 text-green-700 border border-green-100">
                                  {skill}
                                </span>
                              ))
                            ) : (
                              <span className="text-xs text-gray-400">No skills matched</span>
                            )}
                          </div>
                        </div>
                    </div>

                    {candidate.missingSkills.length > 0 && (
                      <div className="mt-3">
                        <p className="text-xs text-gray-400">Missing: {candidate.missingSkills.join(', ')}</p>
                      </div>
                    )}
                  </div>

                  {/* Action Section */}
                  <div className="flex flex-col sm:flex-row items-center justify-center gap-3 pt-4 md:pt-0 md:pl-6 md:border-l border-gray-100">
                    {activeTab === 'pending' ? (
                      <>
                        <Button 
                          variant="secondary" 
                          onClick={() => handleDecision(candidate.id, 'rejected')}
                          className="w-full sm:w-auto text-red-600 border-red-100 hover:bg-red-50"
                        >
                          <XCircle className="h-4 w-4 mr-2 inline" />
                          Reject
                        </Button>
                        <Button 
                          variant="success" 
                          onClick={() => handleDecision(candidate.id, 'selected')}
                          className="w-full sm:w-auto"
                        >
                          <CheckCircle className="h-4 w-4 mr-2 inline" />
                          Select
                        </Button>
                      </>
                    ) : (
                      <div className="flex flex-col items-end gap-2">
                         <span className={`flex items-center text-sm font-medium ${activeTab === 'selected' ? 'text-green-600' : 'text-red-600'}`}>
                           {activeTab === 'selected' ? <CheckCircle className="h-4 w-4 mr-1"/> : <XCircle className="h-4 w-4 mr-1"/>}
                           {activeTab === 'selected' ? 'Candidate Selected' : 'Candidate Rejected'}
                         </span>
                         <Button variant="ghost" className="text-xs" onClick={() => handleDecision(candidate.id, 'pending')}>
                           Undo Decision
                         </Button>
                      </div>
                    )}
                    <button 
                      onClick={() => deleteCandidate(candidate.id)} 
                      className="text-gray-300 hover:text-gray-500 p-2" 
                      title="Remove permanently"
                    >
                      <Trash2 className="h-4 w-4" />
                    </button>
                  </div>

                </div>
              </Card>
            ))
          )}
        </div>

        {candidates.length > 0 && activeTab === 'pending' && (
           <div className="mt-8 text-center">
             <button onClick={clearAllCandidates} className="text-sm text-gray-400 hover:text-red-500 underline">
               Clear all data
             </button>
           </div>
        )}

      </main>
    </div>
  );
}
