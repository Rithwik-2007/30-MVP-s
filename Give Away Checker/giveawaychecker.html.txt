    <script>
        // --- Configuration & Dictionaries ---
        
        const POSITIVE_TERMS = [
            /terms\s*(?:&|and)\s*conditions/i,
            /official\s*rules/i,
            /link\s*in\s*bio/i, // Usually implies structure
            /no\s*purchase\s*necessary/i,
            /sponsored/i,
            /partnership/i,
            /partner/i,
            /ad/i
        ];

        const NEGATIVE_TERMS = [
            /pay\s*shipping/i,
            /shipping\s*fee/i,
            /bank\s*details/i,
            /credit\s*card/i,
            /send\s*money/i,
            /cash\s*app/i,
            /venmo/i,
            /wire\s*transfer/i,
            /kindly/i,
            /whatsapp/i,
            /telegram/i
        ];

        const URGENCY_TERMS = [
            /hurry/i,
            /act\s*fast/i,
            /immediately/i,
            /last\s*chance/i,
            /minutes\s*left/i
        ];

        const WINNER_METHOD_TERMS = [
            /announce/i,
            /winner/i,
            /story/i,
            /stories/i,
            /dm\s*the\s*winner/i
        ];

        // --- Core Logic ---

        function analyzeCredibility() {
            const text = document.getElementById('captionInput').value;
            const isVerified = document.getElementById('check-verified').checked;
            const hasWinners = document.getElementById('check-winners').checked;
            const hasCollab = document.getElementById('check-collab').checked;

            if (text.trim().length < 10) {
                alert("Please paste a valid caption to analyze.");
                return;
            }

            // 1. Signal Extraction
            const extractedSignals = {
                brandMentions: /@[\w._]+/.test(text) || hasCollab, // Regex for mentions or checkbox
                hasTerms: POSITIVE_TERMS.some(regex => regex.test(text)),
                hasDeadline: /ends|deadline|closes|until|date/i.test(text),
                paymentRequest: NEGATIVE_TERMS.some(regex => regex.test(text)),
                winnerClarity: WINNER_METHOD_TERMS.some(regex => regex.test(text)),
                verified: isVerified,
                pastWinners: hasWinners
            };

            // 2. Language Risk Analysis
            const flags = [];
            let languageRiskScore = 0;

            // Check Negative Terms
            NEGATIVE_TERMS.forEach(regex => {
                const match = text.match(regex);
                if (match) {
                    flags.push(`Financial request detected: "${match[0]}"`);
                    languageRiskScore += 25;
                }
            });

            // Check Urgency
            let urgencyCount = 0;
            URGENCY_TERMS.forEach(regex => {
                if (regex.test(text)) urgencyCount++;
            });
            if (urgencyCount > 0) {
                flags.push("High urgency language detected");
                languageRiskScore += 10;
            }

            // Check "DM to claim" pattern (Specific common vector)
            if (/dm\s*(me|us)\s*to\s*claim/i.test(text)) {
                flags.push("Requires user to DM to claim (High Risk)");
                languageRiskScore += 20;
            }

            // 3. Scoring Logic
            let score = 50; // Base Score

            // Boosters
            if (extractedSignals.verified) score += 15;
            if (extractedSignals.pastWinners) score += 10;
            if (extractedSignals.hasTerms) score += 15;
            if (extractedSignals.brandMentions) score += 5;
            if (extractedSignals.hasDeadline) score += 5;
            if (extractedSignals.winnerClarity) score += 5;

            // Penalities
            if (extractedSignals.paymentRequest) score -= 40;
            score -= languageRiskScore;

            // Clamp Score
            score = Math.max(0, Math.min(100, score));

            // Determine Risk Level
            let riskLevel = "Questionable";
            let badgeClass = "bg-yellow-100 text-yellow-800";
            
            if (score >= 70) {
                riskLevel = "Safe";
                badgeClass = "bg-green-100 text-green-800";
            } else if (score < 40) {
                riskLevel = "High Risk";
                badgeClass = "bg-red-100 text-red-800";
            }

            // Generate Reasoning Summary
            let summary = [];
            if (extractedSignals.verified) {
                summary.push("The account is verified, which significantly increases credibility.");
            } else {
                summary.push("The account is unverified.");
            }

            if (extractedSignals.hasTerms) {
                summary.push("Terms and conditions or rules were detected in the text.");
            } else {
                summary.push("No clear reference to official rules or terms was found.");
            }

            if (flags.length > 0) {
                summary.push(`Language analysis flagged ${flags.length} potential risk factors, including ${flags[0].toLowerCase()}.`);
            }

            if (extractedSignals.paymentRequest) {
                summary.push("CRITICAL: The text mentions payment methods or fees, which is highly atypical for legitimate giveaways.");
            }

            // --- Render Output ---
            
            document.getElementById('score-value').innerText = score;
            const badge = document.getElementById('risk-badge');
            badge.innerText = riskLevel;
            badge.className = `px-4 py-2 rounded-full text-sm font-bold ${badgeClass}`;

            // Render Signals
            const signalList = document.getElementById('signal-list');
            signalList.innerHTML = `
                <li class="flex items-center gap-2">
                    ${getIcon(extractedSignals.verified)} 
                    <span>Account Verification</span>
                </li>
                <li class="flex items-center gap-2">
                    ${getIcon(extractedSignals.hasTerms)} 
                    <span>Terms & Conditions Presence</span>
                </li>
                 <li class="flex items-center gap-2">
                    ${getIcon(extractedSignals.winnerClarity)} 
                    <span>Winner Announcement Clarity</span>
                </li>
                <li class="flex items-center gap-2">
                    ${getIcon(!extractedSignals.paymentRequest)} 
                    <span>No Payment Requests</span>
                </li>
            `;

            // Render Flags
            const flagList = document.getElementById('flag-list');
            const noFlagsMsg = document.getElementById('no-flags-msg');
            flagList.innerHTML = '';
            
            if (flags.length > 0) {
                noFlagsMsg.classList.add('hidden');
                flags.forEach(flag => {
                    const li = document.createElement('li');
                    li.innerHTML = `<i class="fa-solid fa-triangle-exclamation mr-2"></i> ${flag}`;
                    flagList.appendChild(li);
                });
            } else {
                noFlagsMsg.classList.remove('hidden');
            }

            // Render Reasoning
            document.getElementById('reasoning-text').innerText = summary.join(" ");

            // Show Result Section
            document.getElementById('results-container').classList.remove('hidden');
            document.getElementById('results-container').scrollIntoView({ behavior: 'smooth', block: 'start' });
        }

        function getIcon(isPositive) {
            return isPositive 
                ? `<i class="fa-solid fa-circle-check text-green-500"></i>` 
                : `<i class="fa-solid fa-circle-xmark text-slate-300"></i>`;
        }

        // --- UI Utils ---

        function switchTab(tab) {
            const textTab = document.getElementById('tab-text');
            const linkTab = document.getElementById('tab-link');
            const textPanel = document.getElementById('input-text-panel');
            const linkPanel = document.getElementById('input-link-panel');

            if (tab === 'text') {
                textTab.className = "flex-1 py-3 text-sm font-medium text-blue-600 border-b-2 border-blue-600 bg-blue-50/50";
                linkTab.className = "flex-1 py-3 text-sm font-medium text-slate-500 hover:text-slate-700";
                textPanel.classList.remove('hidden');
                linkPanel.classList.add('hidden');
            } else {
                linkTab.className = "flex-1 py-3 text-sm font-medium text-blue-600 border-b-2 border-blue-600 bg-blue-50/50";
                textTab.className = "flex-1 py-3 text-sm font-medium text-slate-500 hover:text-slate-700";
                linkPanel.classList.remove('hidden');
                textPanel.classList.add('hidden');
            }
        }

        function resetAnalysis() {
            document.getElementById('results-container').classList.add('hidden');
            document.getElementById('captionInput').focus();
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }
