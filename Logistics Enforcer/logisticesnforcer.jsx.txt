import React, { useState, useRef, useEffect } from 'react';
import { 
  ShieldAlert, 
  Send, 
  AlertTriangle, 
  FileText, 
  Twitter, 
  MessageSquare, 
  Camera, 
  MapPin, 
  Scale, 
  ExternalLink,
  ChevronRight,
  Loader2,
  Volume2,
  Info,
  ShieldCheck
} from 'lucide-react';
import { GeminiService } from './services/geminiService';
import { AppMode, EnforcerResponse, ChatMessage } from './types';
import { VoiceAssistant } from './components/VoiceAssistant';

const gemini = new GeminiService();

const App: React.FC = () => {
  const [input, setInput] = useState('');
  const [image, setImage] = useState<string | null>(null);
  const [response, setResponse] = useState<EnforcerResponse | null>(null);
  const [isProcessing, setIsProcessing] = useState(false);
  const fileInputRef = useRef<HTMLInputElement>(null);

  const handleAnalysis = async (customInput?: string) => {
    const textToAnalyze = customInput || input;
    if (!textToAnalyze && !image) return;

    setIsProcessing(true);
    try {
      const result = await gemini.analyzeSituation(textToAnalyze, image?.split(',')[1]);
      setResponse(result);
    } catch (error) {
      console.error(error);
      alert("Analysis failed. This tool is for draft assistance only. If you are in physical danger, contact local authorities immediately.");
    } finally {
      setIsProcessing(false);
    }
  };

  const handleFileUpload = (e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (file) {
      const reader = new FileReader();
      reader.onload = (event) => setImage(event.target?.result as string);
      reader.readAsDataURL(file);
    }
  };

  const playTTS = async (text: string) => {
    try {
      const base64Audio = await gemini.generateSpeech(text);
      if (base64Audio) {
        // Log to console as a fallback, actual audio would require PCM processing
        console.info("Enforcer Guidance: ", text);
        alert("Enforcer Insight: " + text);
      }
    } catch (e) {
      console.error(e);
    }
  };

  return (
    <div className="min-h-screen flex flex-col items-center p-4 md:p-8">
      {/* Header */}
      <header className="w-full max-w-4xl flex justify-between items-center mb-12">
        <div className="flex items-center gap-3">
          <div className="bg-red-600 p-2 rounded-lg">
            <ShieldAlert className="w-8 h-8 text-white" />
          </div>
          <div>
            <h1 className="text-2xl font-black uppercase tracking-tighter italic">Logistics Enforcer</h1>
            <p className="text-zinc-500 text-xs font-bold uppercase tracking-widest">Defending Indian Consumers</p>
          </div>
        </div>
        <VoiceAssistant onTranscript={(text) => handleAnalysis(text)} />
      </header>

      <main className="w-full max-w-4xl space-y-8">
        {/* Safety Disclaimer Banner */}
        <div className="bg-zinc-900 border border-zinc-800 rounded-xl p-3 flex items-center gap-3 text-[10px] text-zinc-500 uppercase font-bold tracking-widest">
          <Info className="w-4 h-4 text-blue-500" />
          This AI is a drafting tool. It does not provide legal advice. Safety first.
        </div>

        {/* Input Area */}
        <div className="bg-zinc-900 border-2 border-zinc-800 rounded-2xl p-6 shadow-2xl">
          <textarea
            value={input}
            onChange={(e) => setInput(e.target.value)}
            placeholder="Describe the failure: 'Driver asking for extra cash off-app' or 'Package marked delivered but not received'..."
            className="w-full bg-transparent border-none focus:ring-0 text-lg resize-none mb-4 h-32 placeholder-zinc-700"
          />
          
          {image && (
            <div className="relative inline-block mb-4">
              <img src={image} className="h-32 rounded-lg border border-zinc-700" alt="Evidence" />
              <button 
                onClick={() => setImage(null)}
                className="absolute -top-2 -right-2 bg-red-600 rounded-full p-1"
              >
                <AlertTriangle className="w-3 h-3" />
              </button>
            </div>
          )}

          <div className="flex items-center justify-between border-t border-zinc-800 pt-4">
            <div className="flex gap-4">
              <button 
                onClick={() => fileInputRef.current?.click()}
                className="p-2 text-zinc-400 hover:text-white transition-colors"
                title="Attach Evidence Screenshot"
              >
                <Camera className="w-6 h-6" />
                <input type="file" ref={fileInputRef} hidden onChange={handleFileUpload} accept="image/*" />
              </button>
              <div className="flex items-center gap-2 px-3 py-1 bg-zinc-800 rounded-full text-[10px] font-bold text-zinc-500 uppercase tracking-widest">
                <MapPin className="w-3 h-3" /> India Context
              </div>
            </div>
            
            <button
              onClick={() => handleAnalysis()}
              disabled={isProcessing || (!input && !image)}
              className="bg-white text-black px-6 py-2 rounded-xl font-bold hover:bg-red-600 hover:text-white transition-all flex items-center gap-2 disabled:opacity-50"
            >
              {isProcessing ? <Loader2 className="w-5 h-5 animate-spin" /> : <Send className="w-5 h-5" />}
              ENFORCE
            </button>
          </div>
        </div>

        {/* Results Area */}
        {response && (
          <div className="space-y-6 animate-in fade-in slide-in-from-bottom-4 duration-500">
            
            {/* Classification & Analysis */}
            <div className={`p-4 rounded-xl border-l-4 flex items-center justify-between ${
              response.mode === AppMode.MODE_B ? 'bg-red-900/20 border-red-600' : 'bg-blue-900/20 border-blue-600'
            }`}>
              <div className="flex items-center gap-3">
                <AlertTriangle className={`w-6 h-6 ${response.mode === AppMode.MODE_B ? 'text-red-500' : 'text-blue-500'}`} />
                <div>
                  <h3 className="font-black uppercase tracking-widest text-sm">
                    {response.mode === AppMode.MODE_B ? 'Critical: Tactical Advice' : 'Standard Service Failure'}
                  </h3>
                  <p className="text-xs opacity-70 italic font-medium">{response.analysis}</p>
                </div>
              </div>
              <button onClick={() => playTTS(response.analysis)} className="p-2 bg-white/10 rounded-lg hover:bg-white/20">
                <Volume2 className="w-4 h-4" />
              </button>
            </div>

            {/* Warnings Alert */}
            {response.warnings && response.warnings.length > 0 && (
              <div className="bg-amber-900/10 border border-amber-800 rounded-xl p-4">
                <h4 className="text-[10px] font-black uppercase text-amber-500 mb-2 flex items-center gap-2">
                  <AlertTriangle className="w-3 h-3" /> Safety & Legal Warnings
                </h4>
                <ul className="list-disc list-inside text-xs text-amber-200/80 space-y-1">
                  {response.warnings.map((w, i) => <li key={i}>{w}</li>)}
                </ul>
              </div>
            )}

            {/* Content Blocks */}
            <div className="grid md:grid-cols-2 gap-6">
              {/* Primary Instruction Block */}
              <div className="bg-zinc-900 border border-zinc-800 rounded-2xl p-6">
                <h4 className="flex items-center gap-2 font-bold uppercase text-xs tracking-widest text-zinc-500 mb-4">
                  <ChevronRight className="w-4 h-4 text-red-600" />
                  {response.mode === AppMode.MODE_B ? 'Safety Protocol (Immediate)' : 'Draft for App Support'}
                </h4>
                
                {response.mode === AppMode.MODE_B ? (
                  <ul className="space-y-3">
                    {response.safetyProtocol?.map((item, i) => (
                      <li key={i} className="flex gap-3 text-sm font-semibold">
                        <span className="text-red-600">{i + 1}.</span> {item}
                      </li>
                    ))}
                  </ul>
                ) : (
                  <div className="relative group">
                    <div className="text-[10px] font-bold text-blue-500 mb-2 uppercase flex items-center gap-1">
                       <ShieldCheck className="w-3 h-3" /> Suggested Draft - Review Before Use
                    </div>
                    <pre className="bg-black/50 p-4 rounded-lg text-sm whitespace-pre-wrap font-mono text-zinc-300 border border-zinc-800">
                      {response.immediateStep}
                    </pre>
                    <button 
                      onClick={() => navigator.clipboard.writeText(response.immediateStep || '')}
                      className="absolute top-2 right-2 p-1 bg-zinc-800 rounded opacity-0 group-hover:opacity-100 transition-opacity"
                    >
                      <FileText className="w-4 h-4" />
                    </button>
                  </div>
                )}
              </div>

              {/* Secondary Action Block */}
              <div className="bg-zinc-900 border border-zinc-800 rounded-2xl p-6">
                <h4 className="flex items-center gap-2 font-bold uppercase text-xs tracking-widest text-zinc-500 mb-4">
                  <ChevronRight className="w-4 h-4 text-red-600" />
                  {response.mode === AppMode.MODE_B ? 'Negotiation Script' : 'Social Escalation Draft'}
                </h4>
                
                {response.mode === AppMode.MODE_B ? (
                  <div className="space-y-4">
                    <p className="text-[10px] text-zinc-500 uppercase font-bold italic">Tone: Firm, Non-aggressive</p>
                    <div className="italic text-zinc-300 text-sm border-l-2 border-zinc-700 pl-4 py-2">
                      "{response.negotiationScript}"
                    </div>
                  </div>
                ) : (
                  <div className="space-y-4">
                    <div className="flex items-center justify-between p-3 bg-zinc-950 rounded-lg border border-zinc-800">
                      <div className="flex items-center gap-3">
                        <Twitter className="w-5 h-5 text-sky-500" />
                        <span className="text-xs font-bold uppercase tracking-tight">X Blast</span>
                      </div>
                      <button 
                        onClick={() => window.open(`https://twitter.com/intent/tweet?text=${encodeURIComponent(response.escalationStrategy?.twitter || '')}`)}
                        className="text-[10px] bg-sky-600 text-white px-3 py-1 rounded font-bold"
                      >
                        REVIEW & POST
                      </button>
                    </div>
                    <pre className="text-[10px] text-zinc-500 font-mono overflow-hidden h-20 leading-relaxed italic">
                      {response.escalationStrategy?.twitter}
                    </pre>
                  </div>
                )}
              </div>
            </div>

            {/* Legal Disclaimer & Grounding */}
            <div className="bg-zinc-900 border border-zinc-800 rounded-2xl p-6 space-y-4">
              <div className="flex flex-col md:flex-row justify-between gap-6">
                <div className="flex-1">
                  <h4 className="text-xs font-black uppercase tracking-widest text-zinc-500 mb-2">Legal Disclaimer</h4>
                  <p className="text-[10px] font-medium text-zinc-500 leading-relaxed uppercase tracking-wider">
                    {response.disclaimer}
                  </p>
                </div>
                
                {response.groundingLinks && response.groundingLinks.length > 0 && (
                  <div className="flex-1 md:border-l border-zinc-800 md:pl-6">
                    <h4 className="text-xs font-black uppercase tracking-widest text-zinc-500 mb-2">Rules & Legal Refs</h4>
                    <div className="space-y-2">
                      {response.groundingLinks.map((link, i) => (
                        <a 
                          key={i}
                          href={link.web?.uri || link.maps?.uri}
                          target="_blank"
                          className="flex items-center justify-between text-[10px] font-bold text-red-500 hover:underline"
                        >
                          {link.web?.title || link.maps?.title || 'External Link'}
                          <ExternalLink className="w-3 h-3" />
                        </a>
                      ))}
                    </div>
                  </div>
                )}
              </div>
            </div>
            
            {response.mode === AppMode.MODE_A && response.escalationStrategy?.formalNotice && (
               <div className="bg-black border border-zinc-800 rounded-2xl p-6">
                  <h4 className="text-xs font-black uppercase tracking-widest text-zinc-500 mb-4 flex items-center gap-2">
                    <Scale className="w-4 h-4 text-zinc-400" /> Formal Grievance (Suggested Draft)
                  </h4>
                  <pre className="text-[10px] text-zinc-400 whitespace-pre-wrap font-mono leading-relaxed max-h-48 overflow-y-auto">
                    {response.escalationStrategy.formalNotice}
                  </pre>
               </div>
            )}
          </div>
        )}

        {!response && !isProcessing && (
          <div className="text-center py-20">
             <div className="inline-block p-4 bg-zinc-900 rounded-full mb-6">
               <ShieldAlert className="w-12 h-12 text-zinc-700" />
             </div>
             <h2 className="text-xl font-bold text-zinc-500">Incident Intake Protocol</h2>
             <p className="text-zinc-700 text-sm max-w-xs mx-auto mt-2 italic font-medium">Describe your situation or upload a screenshot to generate tactical drafts.</p>
          </div>
        )}
      </main>

      <footer className="w-full max-w-4xl mt-auto pt-12 pb-8 border-t border-zinc-900 text-center">
        <p className="text-[9px] font-bold text-zinc-700 uppercase tracking-[0.2em] leading-loose">
          Logistics Enforcer AI // Suggested Drafts for Educational Use<br/>
          Not Legal Advice // User Responsibility Required // Ref: India CPA 2019
        </p>
      </footer>
    </div>
  );
};

export default App;
